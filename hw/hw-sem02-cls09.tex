\section{FFT}

\subsection{Домашнее задание}
\begin{enumerate}

  \item
    Пусть $p$ --- простое число и $g$ ---
    \href{https://ru.wikipedia.org/wiki/%D0%9F%D0%B5%D1%80%D0%B2%D0%BE%D0%BE%D0%B1%D1%80%D0%B0%D0%B7%D0%BD%D1%8B%D0%B9_%D0%BA%D0%BE%D1%80%D0%B5%D0%BD%D1%8C_(%D1%82%D0%B5%D0%BE%D1%80%D0%B8%D1%8F_%D1%87%D0%B8%D1%81%D0%B5%D0%BB)}{первообразный корень}
    по модулю $p$. Определим \texttt{DFT} размера $p$ обычным образом: $$f_k = \sum_{i=0}^{p-1} x_i \omega_{p}^{k i}$$

    \begin{enumerate}
      \item
        Определим $x^{\prime}_s = x_{g^{-s} \mod p}$, $f^{\prime}_r = f_{g^{r} \mod p}$ и $\Delta_{i} = \omega_{p}^{g^i}$.
        Покажите, что: $$f^{\prime}_{r} = x_{0} + \sum_{s = 0}^{p - 2} x^{\prime}_{s} \Delta_{(r - s) \mod (p - 1)}$$

        \textbf{Решение:}

        \begin{itemize}
          \item По определению:
          $$f'_r = f_{g^r \mod p} = \sum_{i=0}^{p-1} x_i \cdot \omega_p^{(g^r \mod p) \cdot i}$$

          \item Отделим слагаемое $i = 0$:
          $$f'_r = x_0 + \sum_{i=1}^{p-1} x_i \cdot \omega_p^{g^r \cdot i \mod p}$$

          \item Так как $g$ -- первообразный корень по модулю $p$, элементы $\{g^0, g^1, \ldots, g^{p-2}\} \mod p$ -- это в точности $\{1, 2, \ldots, p-1\}$. Поэтому каждое $i \in \{1, \ldots, p-1\}$ единственным образом представимо как $i = g^{-s} \mod p$ для некоторого $s \in \{0, 1, \ldots, p-2\}$.

          \item Подставим $i = g^{-s} \mod p$:
          $$f'_r = x_0 + \sum_{s=0}^{p-2} x_{g^{-s} \mod p} \cdot \omega_p^{g^r \cdot g^{-s} \mod p}$$

          \item Заметим, что $g^r \cdot g^{-s} = g^{r-s}$, и так как порядок $g$ в $(\Z/p\Z)^*$ равен $p - 1$:
          $$g^{r-s} \mod p = g^{(r - s) \mod (p-1)} \mod p$$

          \item Значит:
          $$\omega_p^{g^r \cdot g^{-s} \mod p} = \omega_p^{g^{(r-s) \mod (p-1)}} = \Delta_{(r-s) \mod (p-1)}$$

          \item Подставляя $x'_s = x_{g^{-s} \mod p}$:
          $$f'_r = x_0 + \sum_{s=0}^{p-2} x'_s \cdot \Delta_{(r-s) \mod (p-1)}$$

          ч.т.д.
        \end{itemize}

      \item
        Придумайте, как вычислить $f_k$ за $\O(p \log{p})$ арифметических операций.

        \textbf{Решение:}

        \begin{itemize}
          \item Из пункта (а) следует, что
          $$f'_r = x_0 + \sum_{s=0}^{p-2} x'_s \cdot \Delta_{(r-s) \mod (p-1)}$$

          Сумма $\sum_{s=0}^{p-2} x'_s \cdot \Delta_{(r-s) \mod (p-1)}$ -- это циклическая свёртка двух последовательностей длины $p - 1$: $(x'_0, x'_1, \ldots, x'_{p-2})$ и $(\Delta_0, \Delta_1, \ldots, \Delta_{p-2})$.

          \item Циклическая свёртка длины $N = p - 1$ вычисляется за $\O(N \log N)$:
          \begin{enumerate}
            \item Дополнить обе последовательности нулями до длины $M \geq 2N - 1$, где $M$ -- степень двойки.
            \item Вычислить DFT обеих последовательностей размера $M$.
            \item Поэлементно перемножить результаты.
            \item Вычислить обратное DFT.
            \item Сложить элементы с индексами, совпадающими по модулю $N$ (свернуть обратно до длины $N$).
          \end{enumerate}

          \item После получения свёртки прибавляем $x_0$ к каждому $f'_r$ за $\O(p)$.

          \item Получили все $f'_r = f_{g^r \mod p}$ для $r = 0, \ldots, p-2$, то есть все $f_k$ для $k \in \{1, \ldots, p-1\}$.

          \item Значение $f_0 = \sum_{i=0}^{p-1} x_i$ вычисляем отдельно за $\O(p)$.

          \item Итого: $\O(p \log p)$ арифметических операций.
        \end{itemize}
    \end{enumerate}

  \item
    Придумайте, как свести вычисление \texttt{DFT} последовательности размера $n = n_1 n_2$ к:
    \begin{enumerate}
      \item
        $n_1$ вычислениям \texttt{DFT} от последовательностей размера $n_2$ и
        $\O(n_1^2 n_2)$ дополнительных арифметических операций.
      \item
        $n_1$ вычислениям \texttt{DFT} от последовательностей размера $n_2$,
        $n_2$ вычислениям \texttt{DFT} от последовательностей размера $n_1$ и
        $\O(n_1 n_2)$ дополнительных арифметических операций.
    \end{enumerate}

    \textbf{Решение:}

    \begin{itemize}
      \item Представим индексы: $i = i_1 + n_1 i_2$ (где $0 \leq i_1 < n_1$, $0 \leq i_2 < n_2$) и $k = n_2 k_1 + k_2$ (где $0 \leq k_1 < n_1$, $0 \leq k_2 < n_2$).

      \item Подставим в определение DFT:
      $$f_{n_2 k_1 + k_2} = \sum_{i_1=0}^{n_1-1}\sum_{i_2=0}^{n_2-1} x_{i_1 + n_1 i_2} \cdot \omega_n^{(n_2 k_1 + k_2)(i_1 + n_1 i_2)}$$

      \item Раскроем показатель степени, используя $\omega_n^{n_2} = \omega_{n_1}$, $\omega_n^{n_1} = \omega_{n_2}$, $\omega_n^{n_1 n_2} = 1$:
      $$\omega_n^{(n_2 k_1 + k_2)(i_1 + n_1 i_2)} = \omega_{n_1}^{k_1 i_1} \cdot \omega_n^{k_2 i_1} \cdot \omega_{n_2}^{k_2 i_2}$$

      \item Поэтому:
      $$f_{n_2 k_1 + k_2} = \sum_{i_1=0}^{n_1-1} \omega_{n_1}^{k_1 i_1} \cdot \omega_n^{k_2 i_1} \cdot \underbrace{\sum_{i_2=0}^{n_2-1} x_{i_1 + n_1 i_2} \cdot \omega_{n_2}^{k_2 i_2}}_{G_{i_1, k_2}}$$

      Внутренняя сумма $G_{i_1, k_2}$ -- это DFT размера $n_2$ от последовательности $(x_{i_1}, x_{i_1 + n_1}, x_{i_1 + 2n_1}, \ldots)$ длины $n_2$. Для каждого $i_1$ -- одно DFT размера $n_2$. Всего $n_1$ таких DFT.

      \item \textbf{(a)} После вычисления всех $G_{i_1, k_2}$ для каждой пары $(k_1, k_2)$ считаем:
      $$f_{n_2 k_1 + k_2} = \sum_{i_1=0}^{n_1-1} \omega_{n_1}^{k_1 i_1} \cdot \omega_n^{k_2 i_1} \cdot G_{i_1, k_2}$$
      напрямую за $\O(n_1)$. Всего пар $(k_1, k_2)$: $n_1 \cdot n_2$, значит дополнительных операций $\O(n_1^2 n_2)$.

      \item \textbf{(b)} Определим $H_{i_1, k_2} = \omega_n^{k_2 i_1} \cdot G_{i_1, k_2}$ (twiddle-множители). Вычисление всех $H$ -- это $\O(n_1 n_2)$ операций.

      Тогда:
      $$f_{n_2 k_1 + k_2} = \sum_{i_1=0}^{n_1-1} \omega_{n_1}^{k_1 i_1} \cdot H_{i_1, k_2}$$

      При фиксированном $k_2$ это DFT размера $n_1$ от последовательности $(H_{0, k_2}, H_{1, k_2}, \ldots, H_{n_1-1, k_2})$. Всего $n_2$ таких DFT.

      Итого: $n_1$ DFT размера $n_2$ + $n_2$ DFT размера $n_1$ + $\O(n_1 n_2)$ операций.
    \end{itemize}

  \item
    Вам даны две последовательности $a_i$ и $b_i$ длины $n = 2^m$. Вычислите за $\O(n \log{n})$
    арифметических операций последовательность:
    $$c_k = \sum_{i \oplus j = k} a_i b_j$$

    Вам может пригодится отображение $x_i \rightarrow f_i$ определенное следующим образом:
    $$f_k = \sum_{i = 0}^{n - 1} x_i (-1)^{\text{bit\_count}(k \land i)}$$

    \textit{Подсказка: покажите, что:}
    $$(-1)^{\text{bit\_count}(k \land (i \oplus j))} = (-1)^{\text{bit\_count}(k \land i)} \cdot (-1)^{\text{bit\_count}(k \land j)}$$

  \item
    Найдите количество \texttt{AVL}-деревьев высоты $h$ из $n$ вершин по простому модулю $p$ за:
    \begin{enumerate}
      \item $\O(h n \log{n})$.
      \item $\O(h N(h))$. Здесь $N(h)$ --- максимальное количество вершин в \texttt{AVL} дереве высоты $h$.
    \end{enumerate}
    Считайте, что в поле вычетов по модулю $p$ вам известен
    \href{https://ru.wikipedia.org/wiki/%D0%9F%D0%B5%D1%80%D0%B2%D0%BE%D0%BE%D0%B1%D1%80%D0%B0%D0%B7%D0%BD%D1%8B%D0%B9_%D0%BA%D0%BE%D1%80%D0%B5%D0%BD%D1%8C_%D0%B8%D0%B7_%D0%B5%D0%B4%D0%B8%D0%BD%D0%B8%D1%86%D1%8B}{первообразный корень из единицы}
    достаточной степени.

    \textbf{Решение (a):}

    \begin{itemize}
      \item Обозначим $f[d][k]$ -- количество AVL-деревьев высоты ровно $d$ из $k$ вершин.

      \item Базовые случаи: $f[-1][0] = 1$ (пустое дерево высоты $-1$), $f[0][1] = 1$ (одна вершина).

      \item Рекуррентность: AVL-дерево высоты $d$ -- это корень + левое поддерево + правое поддерево. Высоты поддеревьев отличаются не более чем на $1$, и хотя бы одно имеет высоту $d - 1$. Три случая:
      \begin{enumerate}
        \item Оба поддерева высоты $d - 1$,
        \item Левое высоты $d - 1$, правое высоты $d - 2$,
        \item Левое высоты $d - 2$, правое высоты $d - 1$.
      \end{enumerate}

      \item Запишем через свёртку производящих функций. Пусть $F_d(x) = \sum_k f[d][k] \cdot x^k$. Тогда:
      $$F_d(x) = x \cdot \big(F_{d-1}(x)\big)^2 + 2x \cdot F_{d-1}(x) \cdot F_{d-2}(x)$$

      Множитель $x$ отвечает за корень (одна вершина).

      \item Каждый шаг -- две свёртки многочленов степени $\leq n$, которые вычисляются через DFT за $\O(n \log n)$.

      \item Всего $h$ шагов (от $d = 1$ до $d = h$). Итого: $\O(h \cdot n \log n)$.

      \item Ответ: $f[h][n]$ -- коэффициент при $x^n$ в $F_h(x)$.
    \end{itemize}

    \textbf{Решение (b):}

    \begin{itemize}
      \item Используем ту же рекуррентность, что и в (а), но заметим, что многочлен $F_d(x)$ имеет степень не более $N(d)$ -- максимального числа вершин в AVL-дереве высоты $d$. Это так, потому что $f[d][k] = 0$ при $k > N(d)$.

      \item $N(d) = 2^{d+1} - 1$ (полное бинарное дерево высоты $d$).

      \item На шаге $d$ мы вычисляем свёртки многочленов степени $\leq N(d-1)$. Используем FFT размера $\O(N(d-1))$, что стоит $\O(N(d-1) \cdot \log N(d-1))$.

      \item Оценим суммарную стоимость. Так как $N(d-1) = 2^d - 1$:
      $$\sum_{d=1}^{h} \O(N(d-1) \log N(d-1)) = \sum_{d=1}^{h} \O(2^d \cdot d)$$

      Используем $\sum_{d=1}^{h} d \cdot 2^d = (h - 1) \cdot 2^{h+1} + 2 = \O(h \cdot 2^h) = \O(h \cdot N(h))$.

      \item Итого: $\O(h \cdot N(h))$.
    \end{itemize}

  \item
    Даны строка $s$ длины $n$ и шаблон $p$ длины $m \le n$.
    Алфавит --- \textbf{не} константа.
    \begin{enumerate}
      \item
        Найдите точные вхождения шаблона за $\O(1)$ \texttt{DFT} размера порядка $n$. \\
        \textit{Подсказка: } посчитайте $\sum_{j = 0}^{m - 1} (s_{i+j} - p_j)^2$ для всех $0 \le i \le n - m$.

        \textbf{Решение:}

        \begin{itemize}
          \item Определим $S(i) = \sum_{j=0}^{m-1}(s_{i+j} - p_j)^2$. Тогда $S(i) = 0$ тогда и только тогда, когда шаблон входит в позиции $i$.

          \item Раскроем скобки:
          $$S(i) = \sum_{j=0}^{m-1} s_{i+j}^2 - 2\sum_{j=0}^{m-1} s_{i+j} \cdot p_j + \sum_{j=0}^{m-1} p_j^2$$

          \item Третье слагаемое -- константа $P = \sum_{j} p_j^2$, вычисляется за $\O(m)$.

          \item Первое слагаемое -- $\sum_{j=0}^{m-1} s_{i+j}^2$ -- скользящая сумма квадратов, вычисляется для всех $i$ через префиксные суммы $s_0^2, s_0^2 + s_1^2, \ldots$ за $\O(n)$.

          \item Второе слагаемое: $\sum_{j=0}^{m-1} s_{i+j} \cdot p_j$ -- это корреляция $s$ и $p$. Определим $q_j = p_{m-1-j}$ (реверс шаблона). Тогда:
          $$\sum_{j=0}^{m-1} s_{i+j} \cdot p_j = (s * q)(i + m - 1)$$
          где $*$ -- линейная свёртка. Она вычисляется за $\O(1)$ DFT размера $\O(n)$: DFT($s$), DFT($q$), поэлементное умножение, обратное DFT.

          \item Итого: $\O(1)$ DFT размера порядка $n$ + $\O(n)$ дополнительных операций.
        \end{itemize}

      \item
        Теперь шаблон может содержать вопросительные знаки. Найдите подстроку $s$,
        точно подходящую под шаблон. $\O(n \log{n})$.

        \textbf{Решение:}

        \begin{itemize}
          \item Введём весовую функцию: $w_j = 0$, если $p_j = {?}$, иначе $w_j = 1$.

          \item Определим:
          $$S(i) = \sum_{j=0}^{m-1} w_j \cdot (s_{i+j} - p_j)^2$$

          $S(i) = 0$ тогда и только тогда, когда шаблон (с учётом подстановок) совпадает с подстрокой $s[i \ldots i+m-1]$.

          \item Раскроем:
          $$S(i) = \underbrace{\sum_j w_j \cdot s_{i+j}^2}_{A(i)} - 2 \underbrace{\sum_j w_j \cdot p_j \cdot s_{i+j}}_{B(i)} + \underbrace{\sum_j w_j \cdot p_j^2}_{C}$$

          \item $C$ -- константа, вычисляется за $\O(m)$.

          \item $A(i) = \sum_j w_j \cdot s_{i+j}^2$ -- свёртка последовательности $s^2$ с реверсом $w$. Одно DFT.

          \item $B(i) = \sum_j (w_j \cdot p_j) \cdot s_{i+j}$ -- свёртка $s$ с реверсом $(w \cdot p)$. Одно DFT.

          \item Итого: $\O(1)$ DFT размера $\O(n)$, то есть $\O(n \log n)$.

          \item Ищем $i$, для которого $S(i) = 0$ -- это позиция подходящей подстроки.
        \end{itemize}

      \item
        Шаблон содержит звёздочки и вопросительные знаки. Проверьте,
        подходит ли строка \textbf{целиком} под шаблон. $\O(n \sqrt{n} \log{n})$.

        \textbf{Решение:}

        \begin{itemize}
          \item Склеим подряд идущие звёздочки в одну. Разобьём шаблон по символам ${*}$ на сегменты $p_1, p_2, \ldots, p_t$ (каждый сегмент содержит только обычные символы и ${?}$). Пусть суммарная длина сегментов $L = |p_1| + \ldots + |p_t| \leq m \leq n$.

          \item Сопоставление -- жадное слева направо: найти вхождение $p_1$, затем $p_2$ после конца $p_1$, и так далее. (Если шаблон не начинается с ${*}$, первый сегмент обязан совпадать с префиксом $s$; аналогично для суффикса.)

          \item Разделим сегменты на длинные ($|p_i| > \sqrt{n}$) и короткие ($|p_i| \leq \sqrt{n}$).

          \item Длинные сегменты: их не более $L / \sqrt{n} \leq \sqrt{n}$ штук. Для каждого находим все вхождения в $s$ с помощью пункта (б) за $\O(n \log n)$. Суммарно: $\O(\sqrt{n} \cdot n \log n)$.

          \item Короткие сегменты: обрабатываем жадно. Поддерживаем текущую позицию $pos$ в $s$. Для очередного короткого сегмента длины $l \leq \sqrt{n}$ перебираем позиции $pos, pos+1, \ldots$, проверяя совпадение (с учётом ${?}$) за $\O(l)$. При совпадении сдвигаем $pos$ на $l$ и переходим к следующему сегменту.

          За всё время $pos$ только растёт и не превышает $n$. Каждая неудачная проверка сдвигает $pos$ на $1$, а удачная -- на $l$. Поэтому суммарное число проверок $\leq n + t$, каждая стоит $\O(\sqrt{n})$. Итого для коротких сегментов: $\O(n \sqrt{n})$.

          \item Общая сложность: $\O(\sqrt{n} \cdot n \log n) + \O(n\sqrt{n}) = \O(n \sqrt{n} \log n)$.
        \end{itemize}
    \end{enumerate}

\end{enumerate}
