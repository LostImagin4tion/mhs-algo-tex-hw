\section{Универсальные семейства}

\subsection{Домашнее задание}
\begin{enumerate}
  \item
    Пусть $A$ --- непустое конечное множество и $\mathcal{A} = \{ f : A \rightarrow \F_2^n \}$ --- 2-независимое семейство хеш-функций на нем.
    Обозначим за $L_k$ множество списков элементов из $A$ длины \textbf{ровно} $k$, за $L_{\le k}$ --- длины \textbf{не более} $k$ и за $L$ --- списков любой длины.
    \begin{enumerate}
      \item
        Фиксируем $k$ и введем семейство хеш-функций $\mathcal{H} = \{ h : L_k \rightarrow \F_2^n \}$ следующим образом:
        пусть $c \in \F_2^n$ и $f_0, f_1, \dots, f_{k-1} \in \mathcal{A}$ семплированы случайно независимо.
        Тогда: $$h([ l_0, l_1, \dots, l_{k-1} ]) = c \oplus f_0(l_0) \oplus f_1(l_1) \oplus \dots \oplus f_{k-1}(l_{k-1})$$
        Покажите, что полученное семейство 2-независимое. \\
        \textit{Подсказка: используйте результат с практики и индукцию по $k$.}

        \textbf{Решение:}

        Для любых двух разных списков
        $l = [l_0, \ldots, l_{k-1}]$, $l' = [l'_0, \ldots, l'_{k-1}]$ из $L_k$ нужно получить
        $$\Pr[h(l) = y_1,\; h(l') = y_2] = \frac{1}{2^{2n}} \quad \forall\, y_1, y_2 \in \F_2^n.$$

        Идем по индукции по $k$.

        \textbf{База $k=1$.}
        Имеем $h([l_0]) = c \oplus f_0(l_0)$, причем из $[l_0] \ne [l'_0]$ следует $l_0 \ne l'_0$.
        Тогда
        $$\Pr[c \oplus f_0(l_0) = y_1,\; c \oplus f_0(l'_0) = y_2]$$
        $$= \sum_{a \in \F_2^n}
        \Pr[f_0(l_0)=a,\; f_0(l'_0)=a\oplus y_1\oplus y_2]\cdot \Pr[c=y_1\oplus a].$$
        По 2-независимости $\mathcal A$ первая вероятность равна $\frac{1}{2^{2n}}$, а $\Pr[c=\cdot]=\frac{1}{2^n}$:
        $$\sum_{a \in \F_2^n} \frac{1}{2^{2n}}\cdot \frac{1}{2^n}
        = 2^n \cdot \frac{1}{2^{3n}} = \frac{1}{2^{2n}}.$$

        \textbf{Переход $k-1 \to k$.}
        Положим
        $$h'([l_0,\ldots,l_{k-2}]) = c \oplus f_0(l_0)\oplus \cdots \oplus f_{k-2}(l_{k-2}),$$
        и по индукционному предположению это 2-независимое семейство на $L_{k-1}$.
        Тогда
        $$h([l_0,\ldots,l_{k-1}]) = h'([l_0,\ldots,l_{k-2}]) \oplus f_{k-1}(l_{k-1}).$$
        Обозначим
        $$R=h'([l_0,\ldots,l_{k-2}]), \qquad R'=h'([l'_0,\ldots,l'_{k-2}]).$$

        Разбираем два варианта.

        \begin{itemize}
          \item Если $l_{k-1} \ne l'_{k-1}$, то при фиксированных $R,R'$:
          $$\Pr[f_{k-1}(l_{k-1}) = y_1\oplus R,\; f_{k-1}(l'_{k-1}) = y_2\oplus R' \mid R,R'] = \frac{1}{2^{2n}},$$
          поскольку $f_{k-1}$ независима от всего, что формирует $R,R'$, и сама 2-независима на разных аргументах.
          Усреднение по $R,R'$ ничего не меняет:
          $$\Pr[h(l)=y_1,\; h(l')=y_2] = \frac{1}{2^{2n}}.$$

          \item Если $l_{k-1}=l'_{k-1}$, то префиксы длины $k-1$ различны.
          Пусть $Z=f_{k-1}(l_{k-1})$, тогда
          $$\Pr[R\oplus Z=y_1,\; R'\oplus Z=y_2]
          = \sum_{z\in\F_2^n}\Pr[Z=z]\Pr[R=y_1\oplus z,\;R'=y_2\oplus z].$$
          $Z$ независимо от $(R,R')$, а по индукционному предположению для разных префиксов
          $$\Pr[R=y_1\oplus z,\;R'=y_2\oplus z]=\frac{1}{2^{2n}}.$$
          Значит
          $$\sum_{z\in\F_2^n}\Pr[Z=z]\cdot\frac{1}{2^{2n}}=\frac{1}{2^{2n}}.$$
        \end{itemize}

        Оба случая дают нужную вероятность, следовательно семейство 2-независимо.

      \item
        Теперь аналогично определенное семейство хеш-функций $\mathcal{H} = \{ h : L_{\le k} \rightarrow \F_2^n \}$:
        независимо семплируем $c \in \F_2^n$ и $f_0, f_1, \dots, f_{k-1} \in \mathcal{A}$, и:
        $$h(l) = c \oplus f_0(l_0) \oplus f_1(l_1) \oplus \dots \oplus f_{len(l) - 1}(l_{len(l) - 1})$$
        (тут $len(l) \le k$).
        Покажите, что полученное семейство также 2-независимое. \\
        \textit{Подсказка: не забудьте про пустой список.}

        \textbf{Решение:}

        Нужно проверить, что для любых $l \ne l'$ из $L_{\le k}$:
        $$\Pr[h(l)=y_1,\; h(l')=y_2]=\frac{1}{2^{2n}} \quad \forall\, y_1,y_2\in\F_2^n.$$

        Из 2-независимости $\mathcal A$ сразу следует 1-независимость: для любого $a\in A$
        значение $f(a)$ равномерно в $\F_2^n$.

        Разбор по длинам:
        \begin{itemize}
          \item \textbf{$|l|=|l'|>0$.}
          Это ровно пункт (а): оба списка лежат в одном $L_m$.

          \item \textbf{Один список пустой.}
          Пусть $l=[]$, а $|l'|=m\ge 1$.
          Тогда
          $$h([])=c,\qquad h(l')=c\oplus f_0(l'_0)\oplus\cdots\oplus f_{m-1}(l'_{m-1}).$$
          Обозначим
          $$F=f_0(l'_0)\oplus\cdots\oplus f_{m-1}(l'_{m-1}).$$
          Поскольку хотя бы одно слагаемое (например, $f_0(l'_0)$) равномерно и независимо от остальных,
          $F$ равномерно в $\F_2^n$.
          Поэтому
          $$\Pr[h([])=y_1,\;h(l')=y_2]
          =\Pr[c=y_1,\;F=y_1\oplus y_2]
          =\Pr[c=y_1]\Pr[F=y_1\oplus y_2]
          =\frac{1}{2^{2n}}.$$

          \item \textbf{$0<|l|<|l'|$.}
          Пусть $|l|=m_1$, $|l'|=m_2$, $m_1<m_2$.
          В $h(l')$ есть слагаемое $f_{m_1}(l'_{m_1})$, которого нет в $h(l)$.
          Запишем
          $$h(l') = S \oplus f_{m_1}(l'_{m_1}),\qquad
          S = c \oplus \bigoplus_{j\ne m_1,\;0\le j<m_2} f_j(l'_j).$$
          Тогда
          $$\Pr[h(l)=y_1,\;h(l')=y_2]$$
          $$= \mathbb E_{h(l),S}\!\left[\mathbf 1[h(l)=y_1]\cdot
          \Pr\!\left(f_{m_1}(l'_{m_1})=y_2\oplus S \mid h(l),S\right)\right].$$
          Условная вероятность равна $\frac{1}{2^n}$ (равномерность и независимость $f_{m_1}$), значит
          $$= \frac{1}{2^n}\Pr[h(l)=y_1].$$
          А $\Pr[h(l)=y_1]=\frac{1}{2^n}$, потому что $h(l)=c\oplus(\cdots)$ и $c$ равномерна.
          Итого снова $\frac{1}{2^{2n}}$.
        \end{itemize}

        Во всех случаях получили нужную вероятность, значит семейство 2-независимо.

    \end{enumerate}

  \item
	Равномерной называется хеш-функция, у которой у каждого значения одинаковое количество прообразов.
	Пусть есть пара множеств $A$, $B$ и две равномерные хеш-функции $f : A \rightarrow [0 \dots n-1]$ и
	$g : B \rightarrow [0 \dots n-1]$, где $n$ --- натуральное число.
	Постройте равномерную хеш-функцию с областью значений $[0 \dots n-1]$ для:
	\begin{enumerate}
	  \item Упорядоченных пар $(A, B)$,
	  
    \textbf{Решение:}

    Определим $h(a, b) = (f(a) + g(b)) \mod n$.

    Проверим равномерность через число прообразов значения $v \in [0 \ldots n-1]$:
    $$|h^{-1}(v)| = |\{(a, b) : (f(a) + g(b)) \mod n = v\}| = \sum_{i=0}^{n-1} |f^{-1}(i)| \cdot |g^{-1}((v - i) \mod n)|$$

    Так как $f$ равномерна, $|f^{-1}(i)| = |A|/n$ для всех $i$.
    Так как $g$ равномерна, $|g^{-1}(j)| = |B|/n$ для всех $j$.

    $$= \sum_{i=0}^{n-1} \frac{|A|}{n} \cdot \frac{|B|}{n} = n \cdot \frac{|A| \cdot |B|}{n^2} = \frac{|A| \cdot |B|}{n}$$

    Число прообразов не зависит от $v$, значит $h$ равномерна.

	  \item Неупорядоченных пар $\{ A, A \}$, если $n$ --- нечётное.
	  
    \textbf{Решение:}

    Определим $h(\{a_1, a_2\}) = (f(a_1) + f(a_2)) \mod n$, где $a_1 \neq a_2$.

    Функция корректно определена, так как сложение коммутативно.

    Обозначим $m = |A|/n$ (число прообразов каждого значения у $f$).

    Зафиксируем $v \in [0 \ldots n-1]$ и посчитаем число неупорядоченных пар $\{a_1, a_2\}$ с $h=v$.

    Число упорядоченных пар $(a_1, a_2)$ (возможно $a_1 = a_2$) с $(f(a_1) + f(a_2)) \mod n = v$:
    $$\sum_{i=0}^{n-1} |f^{-1}(i)| \cdot |f^{-1}((v - i) \mod n)| = \sum_{i=0}^{n-1} m \cdot m = nm^2$$

    Из них диагональные ($a_1 = a_2$, то есть $2f(a_1) \equiv v \pmod{n}$):

    Так как $n$ нечётно, $2$ обратим по модулю $n$, поэтому $2f(a_1) \equiv v \pmod{n}$ равносильно $f(a_1) \equiv v \cdot 2^{-1} \pmod{n}$, и таких элементов ровно $m$.

    Число упорядоченных пар с $a_1 \neq a_2$: $nm^2 - m$.

    Число неупорядоченных пар: $\dfrac{nm^2 - m}{2}$.

    Это значение не зависит от $v$, следовательно $h$ равномерна.

	  \item Покажите, что при четном $n$ соответствующей хеш-функции для $\{ A, A \}$ может не существовать.

    \textbf{Решение:}

    Возьмём $A = \{0, 1, \ldots, n-1\}$ и $f = \mathrm{id}$ (тождественная функция). Тогда $f$ равномерна (каждое значение имеет ровно $1$ прообраз).

    Число неупорядоченных пар $\{a_1, a_2\}$ с $a_1 \neq a_2$ равно $\binom{n}{2} = \dfrac{n(n-1)}{2}$.

    Для равномерности хеш-функции $h : \{A, A\} \to [0 \ldots n-1]$ каждое значение должно иметь ровно
    $$\frac{n(n-1)/2}{n} = \frac{n-1}{2}$$
    прообразов.

    При чётном $n$ число $n - 1$ нечётно, поэтому $\dfrac{n-1}{2} \notin \mathbb{Z}$.

    Значит, равномерной хеш-функции в общем случае при четном $n$ может не быть.

 \end{enumerate}

  \item
    Есть семейство хеш-функций $\mathcal{H} = \{ h : K \rightarrow Y \}$ и подмножество
    $X \subseteq K$, $|X| > 1$. Покажите, что, если $\mathcal{H}$ универсальное
    и $|Y| = c |X|^2$, то
    $$\Prb{h \in \mathcal{H}}\left[ \; |h(X)| = |X| \; \right] \ge 1 - \frac{1}{2 c}$$

    Подсказка: вам может пригодиться \href{https://ru.wikipedia.org/wiki/%D0%9D%D0%B5%D1%80%D0%B0%D0%B2%D0%B5%D0%BD%D1%81%D1%82%D0%B2%D0%BE_%D0%9C%D0%B0%D1%80%D0%BA%D0%BE%D0%B2%D0%B0}{неравенство Маркова}.

    \textbf{Решение:}

    Введем случайную величину $C$ --- число коллизий внутри $X$:
    $$C=\sum_{\{x,x'\}\subseteq X}\mathbf 1[h(x)=h(x')],\qquad x\ne x'.$$
    Событие $|h(X)|=|X|$ эквивалентно $C=0$.

    По линейности ожидания:
    $$\mathbb{E}[C] = \sum_{\{x,x'\}\subseteq X}\Pr[h(x)=h(x')].$$
    Универсальность дает
    $$\Pr[h(x)=h(x')]\le \frac{1}{|Y|}\quad (x\ne x'),$$
    значит
    $$\mathbb{E}[C] \le \binom{|X|}{2}\frac{1}{|Y|}
    = \frac{|X|(|X|-1)}{2}\cdot\frac{1}{c|X|^2}
    = \frac{|X|-1}{2c|X|}
    < \frac{1}{2c}.$$

    Применим неравенство Маркова для неотрицательной $C$:
    $$\Pr[C\ge 1]\le \mathbb{E}[C] < \frac{1}{2c}.$$
    Отсюда
    $$\Pr[|h(X)|=|X|]=\Pr[C=0]=1-\Pr[C\ge 1]\ge 1-\frac{1}{2c}.$$

\end{enumerate}
