\section{Универсальные семейства}

\subsection{Домашнее задание}
\begin{enumerate}
  \item
    Пусть $A$ --- непустое конечное множество и $\mathcal{A} = \{ f : A \rightarrow \F_2^n \}$ --- 2-независимое семейство хеш-функций на нем.
    Обозначим за $L_k$ множество списков элементов из $A$ длины \textbf{ровно} $k$, за $L_{\le k}$ --- длины \textbf{не более} $k$ и за $L$ --- списков любой длины.
    \begin{enumerate}
      \item
        Фиксируем $k$ и введем семейство хеш-функций $\mathcal{H} = \{ h : L_k \rightarrow \F_2^n \}$ следующим образом:
        пусть $c \in \F_2^n$ и $f_0, f_1, \dots, f_{k-1} \in \mathcal{A}$ семплированы случайно независимо.
        Тогда: $$h([ l_0, l_1, \dots, l_{k-1} ]) = c \oplus f_0(l_0) \oplus f_1(l_1) \oplus \dots \oplus f_{k-1}(l_{k-1})$$
        Покажите, что полученное семейство 2-независимое. \\
        \textit{Подсказка: используйте результат с практики и индукцию по $k$.}

        \textbf{Решение:}

        \begin{itemize}
          \item Нужно показать, что для любых двух различных списков 
          $l = [l_0, \ldots, l_{k-1}]$ и $l' = [l'_0, \ldots, l'_{k-1}]$ из $L_k$ и любых $y_1, y_2 \in \F_2^n$:
          $$\Pr[h(l) = y_1 \;\wedge\; h(l') = y_2] = \frac{1}{2^{2n}}$$

          \item Докажем индукцией по $k$.

          \item \textbf{База} ($k = 1$): $h([l_0]) = c \oplus f_0(l_0)$.

          Для различных $[l_0] \neq [l'_0]$ имеем $l_0 \neq l'_0$.

          $$\Pr[c \oplus f_0(l_0) = y_1 \;\wedge\; c \oplus f_0(l'_0) = y_2]$$

          Из первого равенства $c = y_1 \oplus f_0(l_0)$. Поскольку $c$ и $f_0$ независимы:

          $$= \sum_{a \in \F_2^n} \Pr[f_0(l_0) = a \;\wedge\; f_0(l'_0) = a \oplus y_1 \oplus y_2] \cdot \Pr[c = y_1 \oplus a]$$

          По 2-независимости $\mathcal{A}$ первый множитель равен $\frac{1}{2^{2n}}$, второй — $\frac{1}{2^n}$:

          $$= \sum_{a \in \F_2^n} \frac{1}{2^{2n}} \cdot \frac{1}{2^n} = 2^n \cdot \frac{1}{2^{3n}} = \frac{1}{2^{2n}} \quad $$ 
          
          Верно

          \item \textbf{Переход} ($k - 1 \to k$):
          
          Пусть $h'([l_0, \ldots, l_{k-2}]) = c \oplus f_0(l_0) \oplus \ldots \oplus f_{k-2}(l_{k-2})$ — 2-независимо по предположению индукции.
          
          Тогда $h([l_0, \ldots, l_{k-1}]) = h'([l_0, \ldots, l_{k-2}]) \oplus f_{k-1}(l_{k-1})$.

          Обозначим $R = h'([l_0, \ldots, l_{k-2}])$, $R' = h'([l'_0, \ldots, l'_{k-2}])$.

          Так как $l \neq l'$, рассмотрим два случая:

          \item \textbf{Случай 1:} $l_{k-1} \neq l'_{k-1}$.

          $$\Pr[R \oplus f_{k-1}(l_{k-1}) = y_1 \;\wedge\; R' \oplus f_{k-1}(l'_{k-1}) = y_2]$$

          Поскольку $f_{k-1}$ независима от $c, f_0, \ldots, f_{k-2}$ (а значит и от $R, R'$),
          зафиксируем $R, R'$ и используем 2-независимость $\mathcal{A}$:

          $$= \mathbb{E}_{R,R'}\left[\Pr[f_{k-1}(l_{k-1}) = y_1 \oplus R \;\wedge\; f_{k-1}(l'_{k-1}) = y_2 \oplus R' \;\mid\; R, R']\right] = \mathbb{E}_{R,R'}\left[\frac{1}{2^{2n}}\right] = \frac{1}{2^{2n}} \quad $$

          Верно

          \item \textbf{Случай 2:} $l_{k-1} = l'_{k-1}$, но $[l_0, \ldots, l_{k-2}] \neq [l'_0, \ldots, l'_{k-2}]$.

          Обозначим $Z = f_{k-1}(l_{k-1}) = f_{k-1}(l'_{k-1})$. Тогда:
          $$\Pr[R \oplus Z = y_1 \;\wedge\; R' \oplus Z = y_2]$$

          Так как $Z$ (определяется $f_{k-1}$) независимо от $R, R'$ (определяются $c, f_0, \ldots, f_{k-2}$):

          $$= \sum_{z \in \F_2^n} \Pr[Z = z] \cdot \Pr[R = y_1 \oplus z \;\wedge\; R' = y_2 \oplus z]$$

          По предположению индукции $h'$ — 2-независимо, а списки $[l_0, \ldots, l_{k-2}]$ и $[l'_0, \ldots, l'_{k-2}]$ различны, поэтому:

          $$= \sum_{z \in \F_2^n} \Pr[Z = z] \cdot \frac{1}{2^{2n}} = \frac{1}{2^{2n}} \quad $$

          ч.т.д

        \end{itemize}

      \item
        Теперь аналогично определенное семейство хеш-функций $\mathcal{H} = \{ h : L_{\le k} \rightarrow \F_2^n \}$:
        независимо семплируем $c \in \F_2^n$ и $f_0, f_1, \dots, f_{k-1} \in \mathcal{A}$, и:
        $$h(l) = c \oplus f_0(l_0) \oplus f_1(l_1) \oplus \dots \oplus f_{len(l) - 1}(l_{len(l) - 1})$$
        (тут $len(l) \le k$).
        Покажите, что полученное семейство также 2-независимое. \\
        \textit{Подсказка: не забудьте про пустой список.}

        \textbf{Решение:}

        \begin{itemize}
          \item Нужно показать: для любых двух различных списков $l \neq l'$ из $L_{\leq k}$ и любых $y_1, y_2 \in \F_2^n$:
          $$\Pr[h(l) = y_1 \;\wedge\; h(l') = y_2] = \frac{1}{2^{2n}}$$

          Заметим, что $\mathcal{A}$ -- 2-независимо, значит оно также 1-независимо (из дз 6, задача 1б), то есть каждая $f \in \mathcal{A}$ распределена равномерно: $\Pr[f(a) = y] = \frac{1}{2^n}$ для любых $a \in A$, $y \in \F_2^n$.

          Рассмотрим три случая.

          \item \textbf{Случай 1:} $|l| = |l'|$, оба непустые.

          Сводится к пункту (а): оба списка принадлежат $L_{|l|}$, и хеш-функция для них определяется одинаково.

          \item \textbf{Случай 2:} один из списков пустой (пусть $l = []$, $l'$ непустой, $|l'| = m \geq 1$).

          $$h([]) = c, \quad h(l') = c \oplus f_0(l'_0) \oplus \ldots \oplus f_{m-1}(l'_{m-1})$$

          Обозначим $F = f_0(l'_0) \oplus \ldots \oplus f_{m-1}(l'_{m-1})$. Так как каждое $f_i(l'_i)$ равномерно распределено в $\F_2^n$ и $f_i$ независимы друг от друга, то $F$ также распределено равномерно.

          $$\Pr[h([]) = y_1 \;\wedge\; h(l') = y_2] = \Pr[c = y_1 \;\wedge\; F = y_1 \oplus y_2]$$

          Так как $c$ независимо от $f_0, \ldots, f_{m-1}$:

          $$= \Pr[c = y_1] \cdot \Pr[F = y_1 \oplus y_2] = \frac{1}{2^n} \cdot \frac{1}{2^n} = \frac{1}{2^{2n}} \quad $$

          Верно

          \item \textbf{Случай 3:} $0 < |l| < |l'|$ (разные ненулевые длины, пусть $|l| = m_1$, $|l'| = m_2$, $m_1 < m_2$).

          $$h(l) = c \oplus f_0(l_0) \oplus \ldots \oplus f_{m_1 - 1}(l_{m_1 - 1})$$
          $$h(l') = c \oplus f_0(l'_0) \oplus \ldots \oplus f_{m_2 - 1}(l'_{m_2 - 1})$$

          Функция $f_{m_1}$ участвует в $h(l')$, но не в $h(l)$.

          Обозначим $S$ -- всё в $h(l')$ кроме $f_{m_1}(l'_{m_1})$:
          $$S = c \oplus \bigoplus_{j \neq m_1,\; 0 \leq j < m_2} f_j(l'_j)$$
          Тогда $h(l') = S \oplus f_{m_1}(l'_{m_1})$.

          Так как $f_{m_1}$ семплирована независимо от $c, f_0, \ldots, f_{m_1-1}, f_{m_1+1}, \ldots$,
          а значит и от $h(l)$ и $S$:

          $$\Pr[h(l) = y_1 \;\wedge\; S \oplus f_{m_1}(l'_{m_1}) = y_2]$$
          $$= \mathbb{E}_{h(l), S}\left[\Pr[f_{m_1}(l'_{m_1}) = y_2 \oplus S \mid h(l), S] \cdot \mathbf{1}[h(l) = y_1]\right]$$

          По равномерности $f_{m_1}$: $\Pr[f_{m_1}(l'_{m_1}) = y_2 \oplus S \mid h(l), S] = \frac{1}{2^n}$.

          $$= \frac{1}{2^n} \cdot \Pr[h(l) = y_1] = \frac{1}{2^n} \cdot \frac{1}{2^n} = \frac{1}{2^{2n}} \quad $$

          Верно

          (Здесь $\Pr[h(l) = y_1] = \frac{1}{2^n}$, так как $h(l) = c \oplus \ldots$ и $c$ равномерно и независимо.)

          \item Во всех случаях получили $\frac{1}{2^{2n}}$, значит семейство 2-независимо. ч.т.д.
        \end{itemize}

    \end{enumerate}

  \item
	Равномерной называется хеш-функция, у которой у каждого значения одинаковое количество прообразов.
	Пусть есть пара множеств $A$, $B$ и две равномерные хеш-функции $f : A \rightarrow [0 \dots n-1]$ и
	$g : B \rightarrow [0 \dots n-1]$, где $n$ --- натуральное число.
	Постройте равномерную хеш-функцию с областью значений $[0 \dots n-1]$ для:
	\begin{enumerate}
	  \item Упорядоченных пар $(A, B)$,
	  
    \textbf{Решение:}

    Определим $h(a, b) = (f(a) + g(b)) \mod n$.

    Покажем равномерность. Для каждого $v \in [0 \ldots n-1]$ подсчитаем число прообразов:
    $$|h^{-1}(v)| = |\{(a, b) : (f(a) + g(b)) \mod n = v\}| = \sum_{i=0}^{n-1} |f^{-1}(i)| \cdot |g^{-1}((v - i) \mod n)|$$

    Так как $f$ равномерна, $|f^{-1}(i)| = |A|/n$ для всех $i$.
    Так как $g$ равномерна, $|g^{-1}(j)| = |B|/n$ для всех $j$.

    $$= \sum_{i=0}^{n-1} \frac{|A|}{n} \cdot \frac{|B|}{n} = n \cdot \frac{|A| \cdot |B|}{n^2} = \frac{|A| \cdot |B|}{n}$$

    Это не зависит от $v$, значит $h$ равномерна.

	  \item Неупорядоченных пар $\{ A, A \}$, если $n$ --- нечётное.
	  
    \textbf{Решение:}

    Определим $h(\{a_1, a_2\}) = (f(a_1) + f(a_2)) \mod n$, где $a_1 \neq a_2$.

    Функция корректно определена, так как сложение коммутативно.

    Обозначим $m = |A|/n$ (число прообразов каждого значения у $f$).

    Для каждого $v \in [0 \ldots n-1]$ подсчитаем число неупорядоченных пар $\{a_1, a_2\}$ с $h = v$.

    Число упорядоченных пар $(a_1, a_2)$ (возможно $a_1 = a_2$) с $(f(a_1) + f(a_2)) \mod n = v$:
    $$\sum_{i=0}^{n-1} |f^{-1}(i)| \cdot |f^{-1}((v - i) \mod n)| = \sum_{i=0}^{n-1} m \cdot m = nm^2$$

    Из них диагональные ($a_1 = a_2$, то есть $2f(a_1) \equiv v \pmod{n}$):

    Так как $n$ нечётно, $2$ обратим по модулю $n$, поэтому $2f(a_1) \equiv v \pmod{n}$ равносильно $f(a_1) \equiv v \cdot 2^{-1} \pmod{n}$, и таких элементов ровно $m$.

    Число упорядоченных пар с $a_1 \neq a_2$: $nm^2 - m$.

    Число неупорядоченных пар: $\dfrac{nm^2 - m}{2}$.

    Это не зависит от $v$, значит $h$ равномерна.

	  \item Покажите, что при четном $n$ соответствующей хеш-функции для $\{ A, A \}$ может не существовать.

    \textbf{Решение:}

    Возьмём $A = \{0, 1, \ldots, n-1\}$ и $f = \mathrm{id}$ (тождественная функция). Тогда $f$ равномерна (каждое значение имеет ровно $1$ прообраз).

    Число неупорядоченных пар $\{a_1, a_2\}$ с $a_1 \neq a_2$ равно $\binom{n}{2} = \dfrac{n(n-1)}{2}$.

    Для равномерности хеш-функции $h : \{A, A\} \to [0 \ldots n-1]$ каждое значение должно иметь ровно
    $$\frac{n(n-1)/2}{n} = \frac{n-1}{2}$$
    прообразов.

    При чётном $n$ число $n - 1$ нечётно, поэтому $\dfrac{n-1}{2} \notin \mathbb{Z}$.

    Значит, равномерная хеш-функция не может существовать. ч.т.д.

 \end{enumerate}

  \item
    Есть семейство хеш-функций $\mathcal{H} = \{ h : K \rightarrow Y \}$ и подмножество
    $X \subseteq K$, $|X| > 1$. Покажите, что, если $\mathcal{H}$ универсальное
    и $|Y| = c |X|^2$, то
    $$\Prb{h \in \mathcal{H}}\left[ \; |h(X)| = |X| \; \right] \ge 1 - \frac{1}{2 c}$$

    Подсказка: вам может пригодиться \href{https://ru.wikipedia.org/wiki/%D0%9D%D0%B5%D1%80%D0%B0%D0%B2%D0%B5%D0%BD%D1%81%D1%82%D0%B2%D0%BE_%D0%9C%D0%B0%D1%80%D0%BA%D0%BE%D0%B2%D0%B0}{неравенство Маркова}.

    \textbf{Решение:}

    \begin{itemize}
      \item Обозначим $C$ --- число коллизий (неупорядоченных пар $\{x, x'\} \subseteq X$, $x \neq x'$, таких что $h(x) = h(x')$):
      $$C = \sum_{\{x, x'\} \subseteq X} \mathbf{1}[h(x) = h(x')]$$

      \item Заметим, что $|h(X)| = |X|$ тогда и только тогда, когда $C = 0$ (нет коллизий).

      \item Оценим матожидание $C$. По линейности матожидания:
      $$\mathbb{E}[C] = \sum_{\{x, x'\} \subseteq X} \Pr[h(x) = h(x')]$$

      Так как $\mathcal{H}$ универсально, для любых $x \neq x'$:
      $$\Pr[h(x) = h(x')] \leq \frac{1}{|Y|}$$

      Число неупорядоченных пар: $\binom{|X|}{2} = \dfrac{|X|(|X|-1)}{2}$.

      Поэтому:
      $$\mathbb{E}[C] \leq \frac{|X|(|X|-1)}{2} \cdot \frac{1}{|Y|} = \frac{|X|(|X|-1)}{2 \cdot c|X|^2} = \frac{|X|-1}{2c|X|} < \frac{1}{2c}$$

      \item Применим неравенство Маркова ($C \geq 0$):
      $$\Pr[C \geq 1] \leq \frac{\mathbb{E}[C]}{1} < \frac{1}{2c}$$

      \item Следовательно:
      $$\Pr[|h(X)| = |X|] = \Pr[C = 0] = 1 - \Pr[C \geq 1] \geq 1 - \frac{1}{2c}$$

      ч.т.д.
    \end{itemize}

\end{enumerate}
