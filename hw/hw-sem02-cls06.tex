\section{Хеширование строк}

\subsection{Домашнее задание}

  \begin{definition} Семейство хеш-функций $\H = \{ h : X \rightarrow Y \}$ называется
    \begin{itemize}
      \item универсальным, если:
        $$\forall x_1, x_2 \in X, x_1 \neq x_2: \Prb{h \in \H}\left[ \; h(x_1) = h(x_2) \; \right] \le \frac{1}{|Y|}$$

      \item $k$-независимым, если 
        для любых различных $x_1, x_2, \cdots, x_k \in X$, для любых, возможно, совпадающих $y_1, y_2, \cdots, y_k \in Y$ выполняется:
        $$\Prb{h \in \H}\left[ \; \bigwedge_{i=1}^k h(x_i) = y_i \; \right] = \frac{1}{|Y|^k}$$
    \end{itemize}
  \end{definition}

\begin{enumerate}

  \item
    \begin{enumerate}
   \item Докажите, что любое 2-независимое семейство хэш-функций является универсальным.

      Решение:

      \begin{itemize}
        \item Пусть $\H = \{ h : X \to Y \}$ -- 2-независимое семейство. 
        Тогда для любых различных $x_1, x_2 \in X$ и любых $y_1, y_2 \in Y$:
        $$\Prb{h \in \H}\left[ \; h(x_1) = y_1 \;\wedge\; h(x_2) = y_2 \; \right] = \frac{1}{|Y|^2}$$

        \item Нужно показать, что $\Prb{h \in \H}\left[ \; h(x_1) = h(x_2) \; \right] \leq \frac{1}{|Y|}$.

        \item Событие $h(x_1) = h(x_2)$ можно разбить на непересекающиеся события по значению совпадения:
        $$\Prb{h}\left[ h(x_1) = h(x_2) \right] = \sum_{y \in Y} \Prb{h}\left[ h(x_1) = y \;\wedge\; h(x_2) = y \right] = $$

        \item Подставим условие 2-независимости (с $y_1 = y_2 = y$):
        $$= \sum_{y \in Y} \frac{1}{|Y|^2} = |Y| \cdot \frac{1}{|Y|^2} = \frac{1}{|Y|}$$

        \item Получили $\Prb{h}\left[ h(x_1) = h(x_2) \right] = \frac{1}{|Y|} \leq \frac{1}{|Y|}$, 
        что является определением универсальности. ч.т.д.
      \end{itemize}

   \item Докажите, что любое $k+1$-независимое семейство хэш-функций является $k$-независимым.

      Решение:

      \begin{itemize}
        \item Пусть $\H = \{ h : X \to Y \}$ -- $(k+1)$-независимое семейство. 
        Тогда для любых различных $x_1, \ldots, x_{k+1} \in X$ и любых $y_1, \ldots, y_{k+1} \in Y$:
        $$\Prb{h \in \H}\left[ \; \bigwedge_{i=1}^{k+1} h(x_i) = y_i \; \right] = \frac{1}{|Y|^{k+1}}$$

        \item Нужно показать, что для любых различных $x_1, \ldots, x_k \in X$ и любых $y_1, \ldots, y_k \in Y$:
        $$\Prb{h \in \H}\left[ \; \bigwedge_{i=1}^{k} h(x_i) = y_i \; \right] = \frac{1}{|Y|^k}$$

        \item Выберем произвольный $x_{k+1} \in X$, отличный от $x_1, \ldots, x_k$ 
        (это возможно, так как $|X| \geq k + 1$, иначе $(k+1)$-независимость не имела бы смысла).
        
        Разобьём событие $\bigwedge_{i=1}^{k} h(x_i) = y_i$ по всем возможным значениям $h(x_{k+1})$:

        $$\Prb{h}\left[ \bigwedge_{i=1}^{k} h(x_i) = y_i \right] = \sum_{y \in Y} \Prb{h}\left[ \bigwedge_{i=1}^{k} h(x_i) = y_i \;\wedge\; h(x_{k+1}) = y \right] =$$

        \item Применим $(k+1)$-независимость:
        $$= \sum_{y \in Y} \frac{1}{|Y|^{k+1}} = |Y| \cdot \frac{1}{|Y|^{k+1}} = \frac{1}{|Y|^k}$$

        \item Получили определение $k$-независимости.
      \end{itemize}

    \end{enumerate}

  \item
    Придумайте строку длины $n$ над алфавитом $\{0, 1\}$, в которой $\Omega(n^2)$ различных подстрок.

  \item
    Даны две строки суммарной длины $n$. Найдите их самую длинную общую подстроку. $\O(n \log n)$.

    Решение:

    \begin{itemize}
      \item Идея: бинарный поиск по длине общей подстроки + полиномиальное хеширование 
      для проверки существования общей подстроки заданной длины.

      \item Если строки $s$ и $t$ имеют общую подстроку длины $L$, 
      то у них есть общая подстрока любой длины $L' \leq L$
      (как префикс этой подстроки). Это позволяет использовать бинарный поиск.

      \item Полиномиальное хеширование строки $s$ длины $m$:
      $$h(s) = \sum_{i=0}^{m-1} s[i] \cdot p^i \mod q$$
      где $p$ -- основание (простое число, большее размера алфавита), $q$ -- модуль (большое простое число).

      Хеш любой подстроки $s[l, r]$ вычисляется за $\O(1)$ по предпосчитанным префиксным хешам:
      $$h(s[l, r]) = \frac{H[r+1] - H[l]}{p^l} \mod q$$
      где $H[i] = \sum_{j=0}^{i-1} s[j] \cdot p^j \mod q$ -- префиксный хеш.

      \item Финальный алгоритм:
      \begin{enumerate}
        \item Предподсчёт: вычислить массивы префиксных хешей $H_s$, $H_t$
        и степени $p^0, p^1, \ldots, p^n$ за $\O(n)$.
        После этого хеш любой подстроки вычисляется за $\O(1)$.

        \item Запустить бинарный поиск по длине $L$ в диапазоне $[0, \min(|s|, |t|)]$.
        На каждой итерации функция $\texttt{check}(L)$ проверяет существование общей подстроки длины $L$:

        \begin{enumerate}
          \item Перебрать все подстроки длины $L$ в строке $s$.
          Для каждой подстроки $s[i, i+L-1]$ вычислить её хеш за $\O(1)$
          (по предподсчитанным $H_s$ и степеням $p$) и добавить в хеш-таблицу.

          \item Перебрать все $|t| - L + 1$ подстрок длины $L$ в строке $t$.
          Для каждой подстроки $t[j, j+L-1]$ вычислить её хеш за $\O(1)$ и проверить, есть ли он в хеш-таблице.

          \item Если совпадение найдено -- вернуть \texttt{true}, иначе -- \texttt{false}.
        \end{enumerate}

        Стоимость одного вызова $\texttt{check}$: $\O(|s| + |t|) = \O(n)$.

        \item Если $\texttt{check}(L)$ возвращает \texttt{true}, ищем значение $L$ больше, иначе -- меньше.
      \end{enumerate}

      \item Сложность по времени:
      \begin{itemize}
        \item Предподсчёт хешей: $\O(n)$
        \item Бинарный поиск: $\O(\log n)$ итераций, каждая за $\O(n)$
      \end{itemize}

      Итого: $\O(n \log n)$

      \item Сложность по памяти:
      \begin{itemize}
        \item Префиксные хеши и степени: $\O(n)$
        \item Хеш-таблица: $\O(n)$
      \end{itemize}

      Итого: $\O(n)$
      
    \end{itemize}

  \item
    Даны строка $s$ длины $n$ и число $k$. Найдите в $s$ за время $\O(n)$ самую длинную подстроку,
    представимую как степень  некоторой строки $x$, для которой $|x| = k$ (степенью строки
    называют её конкатенацию с собой несколько раз).

    Решение:

    \begin{itemize}
      \item Идея: подстрока $s[i, i + mk - 1]$ является $m$-й степенью строки $x = s[i, i+k-1]$,
      когда $m$ последовательных блоков длины $k$, начиная с позиции $i$, совпадают:
      $$s[i, i+k-1] = s[i+k, i+2k-1] = \ldots = s[i+(m-1)k, i+mk-1]$$

      \item Предподсчёт: вычислим префиксные хеши и степени $p$ за $\O(n)$.
      После этого хеш любой подстроки вычисляется за $\O(1)$.

      \item Определим массив $blocks$: для каждой позиции $i$ от $0$ до $n - 2k$
      $$blocks[i] = \begin{cases} 1, & \text{если } h(s[i, i+k-1]) = h(s[i+k, i+2k-1]) \\ 0, & \text{иначе} \end{cases}$$
      То есть $blocks[i] = 1$ означает, что блок длины $k$ с позиции $i$ совпадает с блоком с позиции $i + k$.
      Вычисляется за $\O(n)$ суммарно.

      \item Определим массив $postfixCount$:
      
      $postfixCount[i] = $ максимальное число последовательных единиц в
      $blocks[i], $ \newline $ blocks[i+k], blocks[i+2k], \ldots$
      
      $$postfixCount[i] = \begin{cases} 0, & \text{если } blocks[i] = 0 \text{ или } i > n - 2k \\ postfixCount[i + k] + 1, & \text{если } blocks[i] = 1 \end{cases}$$

      Пройдемся по строке справа налево и вычислим $postfixCount[i]$ по формуле выше за $\O(n)$.

      \item Тогда для начальной позиции $i$ максимальное число копий $x = s[i, i+k-1]$ равно $postfixCount[i] + 1$,
      
      а длина соответствующей подстроки равна $(postfixCount[i] + 1) \cdot k$.

      \item Ответ:
      $$\max_{0 \leq i \leq n - k} (postfixCount[i] + 1) \cdot k$$

      \item Сложность по времени:
      \begin{itemize}
        \item Предподсчёт хешей: $\O(n)$
        \item Вычисление массива $blocks$: $\O(n)$
        \item Вычисление массива $postfixCount$: $\O(n)$
        \item Поиск максимума: $\O(n)$
      \end{itemize}

      Итого: $\O(n)$

      \item Сложность по памяти: $\O(n)$ -- массивы хешей, $blocks$, $postfixCount$.

    \end{itemize}

  \item
    Найти подстроку в тексте. При сравнении строк, если несовпадений символов было не более~$k$, строки считаются равными. $\O(nk \log n)$.

    Решение:

    \begin{itemize}
      \item Пусть дан текст $text$ длины $n$ и строка $str$ длины $m$.
      Нужно для каждой позиции $i$ в тексте проверить $text[i, i+m-1] == str$, допуская не более $k$ несовпадений.

      \item Вычислим префиксные хеши и степени для строк $text$ и $str$ за $\O(n)$.
      После этого можно за $\O(1)$ сравнить любые две подстроки по хешам.

      \item Проверяем индекс $i$ (совпадает ли $text[i, i+m-1]$ с $str$ с учетом $\leq k$ несовпадениями):

      \begin{enumerate}
        \item Пусть $mismatchCount = 0$, $pos = 0$ (текущая позиция сравнения внутри $str$).

        \item Пока $pos < m$:
        \begin{itemize}
          \item Бинарным поиском по длине $L$ ищем наибольший общий префикс
          $text[i + pos, \ldots]$ и $str[pos, \ldots]$: 
          
          \begin{itemize}
            \item проверяем, совпадают ли хеши
              $$h(text[i+pos, \; i+pos+L-1]) == h(str[pos, \; pos+L-1])$$

            \item находим максимальное $L$, при котором они равны

            \item Стоимость: $\O(\log n)$.
          \end{itemize} 

          \item Если $pos + L = m$ -- всё оставшееся совпало, выходим.

          \item Иначе в позиции $pos + L$ нашли несовпадение: $mismatchCount += 1$.

          \item Если $mismatchCount > k$ -- позиция $i$ не подходит, выходим.

          \item Иначе перепрыгиваем несовпавший символ: $pos += L + 1$.
        \end{itemize}

        \item Если $mismatchCount \leq k$ -- позиция $i$ подходит.
      \end{enumerate}

      \item Повторяем проверку для каждой позиции $i = 0, 1, \ldots, n - m$.

      \item Сложность по времени:
      \begin{itemize}
        \item Предподсчёт хешей: $\O(n)$

        \item Проверка одного индекса $i$: не более $(k + 1)$ бинарных поисков
        (после $k+1$ несовпадений алгоритм останавливается), каждый за $\O(\log n)$.
        Итого на один индекс: $\O(k \log n)$.

        \item Всего позиций: $n - m + 1 \leq n$.
      \end{itemize}

      Итого: $\O(n) + n \cdot \O(k \log n) = \O(nk \log n)$

      \item Сложность по памяти: $\O(n)$ -- массивы хешей и степеней.

    \end{itemize}

  \item
    Дана строка длины $n$. Запросы:
    \begin{itemize}
      \item \texttt{set(i, c)} -- установить символ на $i$-й позиции в $c$.
      \item \texttt{test(i, j, l)} -- проверить на равенство подстроки длины $l$ начинающиеся в $i$-й и $j$-й позициях.
    \end{itemize}
    $\O(\log{n})$ на запрос, online.

\end{enumerate}
